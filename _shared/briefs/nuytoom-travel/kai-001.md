### 3. Create `src/types/trip.ts`
```typescript
export interface Traveler {
  name: string
  age: number
}

export type TripStyle = 'family' | 'adventure' | 'relaxed' | 'culture'
export type AccommodationLevel = 'budget' | 'standard' | 'luxury'

export interface TripInput {
  travelers: Traveler[]
  totalBudget: number
  destination: string
  departDate: string
  returnDate: string
  preferences: {
    tripStyle: TripStyle
    accommodation: AccommodationLevel
  }
}

export interface DayActivity {
  time: string
  activity: string
  location: string
  estimatedCost: number
  notes?: string
}

export interface DayPlan {
  day: number
  date: string
  accommodation: string
  activities: DayActivity[]
  meals: { type: 'เช้า' | 'กลางวัน' | 'เย็น'; place: string; estimatedCost: number }[]
  dailyTotal: number
}

export interface BudgetBreakdown {
  accommodation: number
  food: number
  activities: number
  transportation: number
  misc: number
  total: number
}

export interface TripPlan {
  title: string
  destination: string
  travelers: Traveler[]
  departDate: string
  returnDate: string
  totalDays: number
  days: DayPlan[]
  budget: BudgetBreakdown
  transportationNotes: string
  tips: string[]
}
```

### 5. Create `src/app/api/generate-plan/route.ts`
```typescript
import { NextRequest, NextResponse } from 'next/server'
import Anthropic from '@anthropic-ai/sdk'
import type { TripInput, TripPlan } from '@/types/trip'

const client = new Anthropic()

export async function POST(request: NextRequest) {
  try {
    const input: TripInput = await request.json()
    const { travelers, totalBudget, destination, departDate, returnDate, preferences } = input

    const totalDays = Math.ceil(
      (new Date(returnDate).getTime() - new Date(departDate).getTime()) / (1000 * 60 * 60 * 24)
    ) + 1

    const travelerList = travelers.map(t => `${t.name} (อายุ ${t.age} ปี)`).join(', ')

    const styleMap: Record<string, string> = {
      family: 'เน้นกิจกรรมครอบครัว เหมาะทุกช่วงอายุ',
      adventure: 'เน้นผจญภัย กีฬา กลางแจ้ง',
      relaxed: 'พักผ่อน ชิลล์ ไม่รีบ',
      culture: 'เน้นวัฒนธรรม ประวัติศาสตร์ อาหารท้องถิ่น',
    }

    const accommodationMap: Record<string, string> = {
      budget: 'ประหยัด (โฮสเทล/เกสต์เฮาส์)',
      standard: 'มาตรฐาน (3 ดาว)',
      luxury: 'หรูหรา (4-5 ดาว)',
    }

    const prompt = `คุณเป็นผู้เชี่ยวชาญวางแผนท่องเที่ยวสำหรับครอบครัวชาวไทย

ข้อมูลทริป:
- ผู้เดินทาง ${travelers.length} คน: ${travelerList}
- งบประมาณรวม: ${totalBudget.toLocaleString('th-TH')} บาท
- ปลายทาง: ${destination}
- วันเดินทาง: ${departDate} ถึง ${returnDate} (${totalDays} วัน)
- สไตล์: ${styleMap[preferences.tripStyle]}
- ที่พัก: ${accommodationMap[preferences.accommodation]}

สร้างแผนท่องเที่ยวละเอียดเป็น JSON ตามโครงสร้าง:
{
  "title": "ชื่อทริป",
  "destination": "${destination}",
  "travelers": ${JSON.stringify(travelers)},
  "departDate": "${departDate}",
  "returnDate": "${returnDate}",
  "totalDays": ${totalDays},
  "days": [
    {
      "day": 1,
      "date": "DD/MM/YYYY",
      "accommodation": "ชื่อที่พักและย่าน",
      "activities": [{"time": "09:00", "activity": "ชื่อ", "location": "สถานที่", "estimatedCost": 500, "notes": "หมายเหตุ"}],
      "meals": [
        {"type": "เช้า", "place": "ร้าน", "estimatedCost": 300},
        {"type": "กลางวัน", "place": "ร้าน", "estimatedCost": 500},
        {"type": "เย็น", "place": "ร้าน", "estimatedCost": 800}
      ],
      "dailyTotal": 3000
    }
  ],
  "budget": {"accommodation": 0, "food": 0, "activities": 0, "transportation": 0, "misc": 0, "total": 0},
  "transportationNotes": "รายละเอียดเที่ยวบิน/รถ/เรือ ขาไปและขากลับ",
  "tips": ["เคล็ดลับ 1", "เคล็ดลับ 2"]
}

กฎสำคัญ: budget.total ต้องไม่เกิน ${totalBudget} บาท | ตอบ JSON ล้วนๆ ไม่มี markdown`

    const message = await client.messages.create({
      model: 'claude-sonnet-4-6',
      max_tokens: 8000,
      messages: [{ role: 'user', content: prompt }],
    })

    const content = message.content[0]
    if (content.type !== 'text') throw new Error('Unexpected response type')

    let plan: TripPlan
    try {
      plan = JSON.parse(content.text)
    } catch {
      const jsonMatch = content.text.match(/\{[\s\S]*\}/)
      if (!jsonMatch) throw new Error('Could not parse plan JSON')
      plan = JSON.parse(jsonMatch[0])
    }

    return NextResponse.json({ plan })
  } catch (error) {
    console.error('Error generating plan:', error)
    return NextResponse.json(
      { error: 'ไม่สามารถสร้างแผนเที่ยวได้ กรุณาลองใหม่อีกครั้ง' },
      { status: 500 }
    )
  }
}
```
